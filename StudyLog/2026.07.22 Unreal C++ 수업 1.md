### Montage
- 애니메이션이 움직이는대로 메시도 같이 움직여야하면, 애니메이션 시퀀스에서 루트모션 켜기

```cpp file:ActionCharacter.cpp
void AActionCharacter::OnRollAction(const FInputActionValue& Value)
{
	if (!RollMontage.IsValid()) return;

	// 스테미너 모자라면 못구름
	if (IStaminaInterface::Execute_GetCurrentStamina(this) < RollStaminaUsage)  return;

	// AnimInstance는 기본적으로 mesh가 가지고 있다.
	// BeginPlay에서 가져오는걸 시도해보고, 혹시 없다면 재시도
	if (!AnimInstance)
	{
		AnimInstance = GetMesh()->GetAnimInstance();
	}

	// 다른 몽타주가 재생중이면 스킵
	if (!AnimInstance->IsAnyMontagePlaying())
	{
		// 구르기 직전에 구를 방향으로 돌리기
		// 이것만 있으면 입력이 없는 경우 무조건 정면으로 회전해버림
		// 입력이 없는경우 last input이 0,0 이니까 회전이 없어서 무조건 X축으로 구른다
		// SetActorRotation(GetLastMovementInputVector().Rotation());
		// UE_LOG(LogTemp, Log, TEXT("%f, %f"), GetLastMovementInputVector().X, GetLastMovementInputVector().Y);
		
		// 이게 있어야 실제 유효한 회전을 넣음
		// 입력이 없어도 현재 바라보는 방향으로 구른다.
		if (!GetLastMovementInputVector().IsNearlyZero())
		{
			SetActorRotation(GetLastMovementInputVector().Rotation());
		}

		PlayAnimMontage(RollMontage);
		IStaminaInterface::Execute_ConsumeStamina(this, RollStaminaUsage);
	}
}
```

```cpp file:StaminaInterface.gen.cpp
bool IStaminaInterface::Execute_ConsumeStamina(UObject* O, float InAmount)
{
	// UObject가 NULL이면 종료
	check(O != NULL);
	// UObject가 UInterface를 구현하지 않았으면 종료
	check(O->GetClass()->ImplementsInterface(UStaminaInterface::StaticClass()));
	StaminaInterface_eventConsumeStamina_Parms Parms;
	
	// Reflection System에서 UFunction을 찾아옴
	UFunction* const Func = O->FindFunction(NAME_UStaminaInterface_ConsumeStamina);
	
	// 있는경우 해당 UFunction을 실행
	// Reflection System을 통해서 실행
	if (Func)
	{
		Parms.InAmount=std::move(InAmount);
		// ProcessEvent에서 BP Overide가 있으면 그걸 실행 (BP Override가 우선)
		// 없으면 _Implementation() 실행 이 과정의 근거를 모르겠음 심연에 빠질까자 0722 07:11에 중단
		O->ProcessEvent(Func, &Parms);
	}
	// 사실상 여기로 오지 않는데, 예외적인 상황에서는 native 함수 호출
	// 해당 인터페이스의 실제 주소를 가져오는데 성공하면
	else if (auto I = (IStaminaInterface*)(O->GetNativeInterfaceAddress(UStaminaInterface::StaticClass())))
	{
		// 해당 함수의 cpp에 구현된 _Implementatation 함수 실행
		Parms.ReturnValue = I->ConsumeStamina_Implementation(InAmount);
	}
	return Parms.ReturnValue;
}
```
