터졌을 때 크래시 내용을 살펴보면
`"Do not directly call Event functions in Interfaces. Call Execute_GetMaxHealth instead."` 인터페이스에서 함수를 직접 호출하는건 금지되어 있는 듯 하다. 
크래쉬를 발생시칸 코드는 다음과 같다.

```cpp file:HealthInterface.gen.cpp
float IHealthInterface::GetMaxHealth() const
{
	// 여기로 들어오면 무조건 Assertion 시킨다.
    check(0 && "Do not directly call Event functions in Interfaces. Call Execute_GetMaxHealth instead.");
    HealthInterface_eventGetMaxHealth_Parms Parms;
    return Parms.ReturnValue;

}

static FName NAME_UHealthInterface_GetMaxHealth = FName(TEXT("GetMaxHealth"));

// 여기로 들어와야 정상 동작
float IHealthInterface::Execute_GetMaxHealth(const UObject* O)
{
	// UObject가 유효한지 확인
    check(O != NULL);
    // UObject가 인터페이스를 구현 했는지 확인
    check(O->GetClass()->ImplementsInterface(UHealthInterface::StaticClass()));
    HealthInterface_eventGetMaxHealth_Parms Parms;
    
    // ReflectionSystem에서 해당 함수를 찾기
    UFunction* const Func = O->FindFunction(NAME_UHealthInterface_GetMaxHealth);

	// 함수가 있으면 
    if (Func)
    {
	    // ProcessEvent가 const가 아니기 때문에 떼고 사용
        const_cast<UObject*>(O)->ProcessEvent(Func, &Parms);
    }

	// 없으면 인테페이스의 주소를 찾아와옴
    else if (auto I = (const IHealthInterface*)(O->GetNativeInterfaceAddress(UHealthInterface::StaticClass())))
    {
	    // 함수의 실행 값을 매개변수에 넣음
        Parms.ReturnValue = I->GetMaxHealth_Implementation();
    }
    
    // 매개 변수를 반환
    return Parms.ReturnValue;
}
```

```cpp file:ActionCharacter.cpp hlt:12,31,37,53,70,72
// 구를 때 스테미나 소비
void AActionCharacter::OnRollAction(const FInputActionValue& Value)
{
	if (!RollMontage) return;

	if (!AnimInstance)
	{
		AnimInstance = GetMesh()->GetAnimInstance();
	}

	// 스테미너 모자라면 못구름
	if (IStaminaInterface::Execute_GetCurrentStamina(this) < RollStaminaUsage)  return;

	// 다른 몽타주가 재생중이면 스킵
	if (!AnimInstance->IsAnyMontagePlaying())
	{
		// 구르기 직전에 구를 방향으로 돌리기
		// 이것만 있으면 입력이 없는 경우 무조건 정면으로 회전해버림
		// 입력이 없는경우 last input이 0,0 이니까 회전이 없어서 무조건 X축으로 구른다
		// SetActorRotation(GetLastMovementInputVector().Rotation());
		// UE_LOG(LogTemp, Log, TEXT("%f, %f"), GetLastMovementInputVector().X, GetLastMovementInputVector().Y);
		
		// 이게 있어야 실제 유효한 회전을 넣음
		// 입력이 없어도 현재 바라보는 방향으로 구른다.
		if (!GetLastMovementInputVector().IsNearlyZero())
		{
			SetActorRotation(GetLastMovementInputVector().Rotation());
		}

		PlayAnimMontage(RollMontage);
		IStaminaInterface::Execute_ConsumeStamina(this, RollStaminaUsage);
	}
}


// 달릴 때 스테미나 소비
void AActionCharacter::ConsumeSprintStamina(float DeltaTime)
{
	// 달리는 상태 + 이동속도가 걷기보다 빠르면 + 몽타주 재생중이 아니면
	bool Condition = bRun
		&& GetMovementComponent()->Velocity.Size() >= WalkSpeed
		&& !AnimInstance->IsAnyMontagePlaying();

	if (Condition)
	{
		// 스테미나 소비 실패시 스프린트 종료
		if (!IStaminaInterface::Execute_ConsumeStamina(this, DeltaTime * SprintStaminaUsagePerSec))
		{
			OnSprintEnd();
		}
	}
}

// 자동 회복
void AActionCharacter::StaminaAutoRecovery(float DeltaTime)
{
	StaminaAutoRecoveryTimer -= DeltaTime;

	if (StaminaAutoRecoveryTimer <= 0)
	{
		if (!FMath::IsNearlyEqual(CurrentStamina, MaxStamina))
			IStaminaInterface::Execute_RecoveryStamina(this, DeltaTime * AutoStaminaRecoveryPerSec);
	}
}

// Called every frame
void AActionCharacter::Tick(float DeltaTime)
{
	Super::Tick(DeltaTime);


	ConsumeSprintStamina(DeltaTime);
	
	StaminaAutoRecovery(DeltaTime);

	auto Val = FMath::FInterpTo(StaminaBar->GetPercent(), CurrentStamina/MaxStamina, DeltaTime, 5);
	StaminaBar->SetPercent(Val);
}
```

```cpp file:ActionCharacter.cpp
bool AActionCharacter::ConsumeStamina_Implementation(float InAmount)
{
	bool bResult = false;
	if (CurrentStamina >= InAmount)
	{
		CurrentStamina -= InAmount;
		bResult = true;

		//타이머로 대체
		//StaminaAutoRecoverySec = StaminaAutoRecoveryCoolTime;
		//타이머로 간단하게 대체 가능
		UWorld* World = GetWorld();
		FTimerManager& TimerManager = World->GetTimerManager();

		// 같은 핸들로 여러번 실행하면 덮어쓰기임
		// StaminaAutoRecoveryCoolTime초 뒤에
		// StaminaAutoRecovery를 실행하고
		// 그 이후로 AutoStaminaRecoveryInterval마다
		// StaminaAutoRecovery 실행
		TimerManager.SetTimer(
			AutoRecorveryStaminaTimer,
			this,
			&AActionCharacter::StaminaAutoRecovery,
			AutoStaminaRecoveryInterval,
			true,
			StaminaAutoRecoveryCoolTime
		);

	}

	CurrentStamina = FMath::Clamp(CurrentStamina, 0, MaxStamina);
	//UE_LOG(LogTemp, Log, TEXT("현재 Stamina : %.1f"), CurrentStamina);
	return bResult;
}
```
