```cpp file:StaHolder.gen.cpp hlt:14
UStatComponent* IStatHolder::Execute_GetStatComponent(UObject* O)
{
	// UObject가 없으면 종료
	check(O != NULL);
	
	// UObject가 해당 인터페이스를 구현 했는지 확인
	check(O->GetClass()->ImplementsInterface(UStatHolder::StaticClass()));
	
	// 함수 실행 결과를 담을 변수
	StatHolder_eventGetStatComponent_Parms Parms;
	
	//여기서 해당 UObject의 함수를 찾아온다.
	//그럼 플루프린트랑, C++에서 둘다 구현되어 있으면 어쩌지?의 의문이 여기서 해결
	UFunction* const Func = O->FindFunction(NAME_UStatHolder_GetStatComponent);
	if (Func)
	{
		// 찾은 함수를 ProcessEvent를 통해서 실핼
		// ProcessEvent(Func) => Func->Invoke()
		O->ProcessEvent(Func, &Parms);
	}
	
	// 함수를 못 찾았을 때 사실 예외처리용으로 네이티브 함수 호출해줌
	else if (auto I = (IStatHolder*)(O->GetNativeInterfaceAddress(UStatHolder::StaticClass())))
	{
		Parms.ReturnValue = I->GetStatComponent_Implementation();
	}
	return Parms.ReturnValue;
}
```

```cpp file:class.cpp hlt:1,72-75
UFunction* UClass::FindFunctionByName(FName InName, EIncludeSuperFlag::Type IncludeSuper) const
{
	LLM_SCOPE(ELLMTag::UObject);
	LLM_SCOPE_BYTAG(UObject_UClass);

	UFunction* Result = nullptr;

	UE_AUTORTFM_OPEN
	{
		// 부모 클래스를 가져옴
		// 여기서는 AActionChracter
		UClass* SuperClass = GetSuperClass();
		// 기본값이 부모 포함
		// 부모 클래스 포함 안하기 || (구현한 인터페이스 없고, 부모 클래스가 없을 때)
		// BP클래스에서 cpp 부모의 인터페이스 함수를 단순 오버라이드만 했다면 Interface는 비어있게 된다/
		if (IncludeSuper == EIncludeSuperFlag::ExcludeSuper || ( Interfaces.Num() == 0 && SuperClass == nullptr ) )
		{
			// 해당 클래스의 함수 맵만 찾아보고, 캐싱은 하지 않기
			// Trivial case: just look up in this class's function map and don't involve the cache
			FUClassFuncScopeReadLock ScopeLock(FuncMapLock);
			Result = FuncMap.FindRef(InName);
		}
		else
		{
			// Check the cache
			bool bFoundInCache = false;
			{
				// 일종의 lock인데
				// Read는 여러명이 가능 하지만, 한 명이라도 Read하면 Write불가
				// 내부 코드를 살펴보니 Read는 Sharedlock을 사용함
				// 윈도우 API에 Mutex를 사용 하는 듯
				// 또한 RAII 패턴으로 객체 생성 시 락을 걸고 소멸 시 락이 사라진다.
				// 여기 중괄호 빠져 나가면 사라짐
				FUClassFuncScopeReadLock ScopeLock(AllFunctionsCacheLock);
				
				// 캐싱된 함수에 현재 찾는 함수가 있는지 확인
				if (UFunction** SuperResult = AllFunctionsCache.Find(InName))
				{
					Result = *SuperResult;
					bFoundInCache = true;
				}
			}

			// 캐싱이 되지 않았다면
			if (!bFoundInCache)
			{
				// Try this class's FuncMap first
				{
					// 락을 걸고 현재 클래스 내부의 함수 맵을 찾음
					FUClassFuncScopeReadLock ScopeLock(FuncMapLock);
					Result = FuncMap.FindRef(InName);
				}

				// 있는 경우
				if (Result)
				{
					// Cache the result
					// 쓰기 락걸고 함수 캐싱
					FUClassFuncScopeWriteLock ScopeLock(AllFunctionsCacheLock);
					AllFunctionsCache.Add(InName, Result);
				}
				else
				{
					// 없는 경우
					
					// 구현한 인터페이스 있는지 확인 
					/*
					The list of interfaces which this class implements, 
					along with the pointer property that is located at the offset of the interface's vtable.
					If the interface class isn't native, the property will be null.
					*/
					// 위 설명은 읽어보면 구현된 Interfaces 내부에는 해당 c++ 인터페이스가 구현한 인터페이스가 들어간다.
					// 구조체 내용중에 BP에 의해 오버라이드 확인하는 bool이 있다.
					//void FKismetCompilerContext::AddInterfacesFromBlueprint(UClass* Class) 이 함수에서 등록하는듯
					// 만약 이 인터페이스가 BP에서 선언 된것이면 null이 된다.
					// Check superclass and interfaces
					// 간단하게 이 클래스에서 새로 상속받은 인터페이스 있는지
					if (Interfaces.Num() > 0)
					{
						// 구현된 인터페이스를 돌면서 
						for (const FImplementedInterface& Inter : Interfaces)
						{
							// 해당 인터페이스들을 돌면서 인터페이스가 유효하면 함수 찾기(인테페이스에서)
							Result = Inter.Class ? Inter.Class->FindFunctionByName(InName) : nullptr;
							if (Result)
							{
								break;
							}
						}
					}
					
					// 부모 클래스 있으면 거기서 먼저 찾아보기
					UFunction* SuperResult = nullptr;
					if (SuperClass != nullptr)
					{
						SuperResult = SuperClass->FindFunctionByName(InName);
					}

					// Check for multiple inherit: If a superclass implements the same interface, use its implementation instead
					// 이 클래스에 구현 없기 || (부모에 동일한 인터페이스 구현 있으면 )
					// 부모의 구현 사용하기 ? 왜 그런것이지 물어보자
					if (!Result || (SuperResult && SuperResult->GetOwnerClass()->ImplementsInterface(Result->GetOwnerClass())))
					{
						Result = SuperResult;
					}

					{
						// Do a final check to make sure the function still doesn't exist in this class before we add it to the cache, in case the function was added by another thread since we last checked
						// This avoids us writing null (or a superclass func with the same name) to the cache if the function was just added
						// 함수를 캐식하기 전에 Readlock을 걸어서 실제로 있는지 확인
						FUClassFuncScopeReadLock ScopeLockFuncMap(FuncMapLock);
						if (FuncMap.FindRef(InName) == nullptr)
						{
							// Cache the result (even if it's nullptr)
							// 위 문장은 이 클래스에는 구현이 없어도 (부모에만 있어도) 캐싱한다는 뜻
							FUClassFuncScopeWriteLock ScopeLock(AllFunctionsCacheLock);
							AllFunctionsCache.Add(InName, Result);
						}
					}
				}
			}
		}
	};

	return Result;
}
```
